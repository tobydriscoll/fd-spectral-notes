var('M g')

(M, g)

xi(s) = 1 +(2/M)*tan(g*(s-1)/2)

diff(xi)

s |--> ((tan(1/2*g*(s - 1))^2 + 1)*g/M)

simplify(diff(xi,s,s)(s))

(tan(1/2*g*(s - 1))^2 + 1)*g^2*tan(1/2*g*(s - 1))/M

var('x')

x

s(x) = 1 + (2/g)*atan(M*(x-1)/2)

diff(s,x)

x |--> 4*M/((M^2*(x - 1)^2 + 4)*g)

diff(s,x,x)

x |--> -8*M^3*(x - 1)/((M^2*(x - 1)^2 + 4)^2*g)
